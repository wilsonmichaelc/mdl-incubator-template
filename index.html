#!/usr/bin/python

import os, sys, subprocess, time, shutil, re, json

while True:
    try:
        job = fetchJob()
        if job is None:
            print("No Jobs Found. Sleeping 10 seconds.")
            time.sleep(10)
        else:
            print("Executing Job: %d" % job.id)
            updateStatus(job.id, "running")
            job = executeJob(job)
            if job.results is "error":
                updateStatus(job.id, "error")
                print("Job failed.")
            else:
                updateResult(job.id, json.dumps(job.results))
                updateStatus(job.id, "done")
                print("Job complete.")
    except Exception:
        pass
